import React, { useState } from 'react';

// A simple React functional component to test MDX integration.
// This component displays a greeting and includes a small interactive counter.
const App = ({ name = 'Guest' }) => {
  // Use React's useState hook to manage a simple counter.
  // This demonstrates basic state management within a component.
  const [count, setCount] = useState(0);

  // Function to increment the counter
  const incrementCount = () => {
    setCount(prevCount => prevCount + 1);
  };

  return (
    <div
      className="p-6 mt-4 border-2 border-green-600 rounded-lg bg-green-50 text-green-800 font-sans shadow-md flex flex-col items-center justify-center space-y-4"
    >
      <h1 className="text-2xl font-bold">Hello, {name}!</h1>
      <p className="text-lg">
        This is a React component rendered within your MDX file.
      </p>
      <div className="flex items-center space-x-3">
        <p className="text-md">
          You've clicked the button <span className="font-bold text-xl">{count}</span> times.
        </p>
        <button
          onClick={incrementCount}
          className="px-4 py-2 bg-green-700 text-white rounded-md shadow-lg hover:bg-green-800 transition-colors duration-200 ease-in-out font-semibold"
        >
          Click Me!
        </button>
      </div>
      <p className="text-sm text-gray-600">
        Try changing the `name` prop in your MDX file!
      </p>
    </div>
  );
};

export default App;
